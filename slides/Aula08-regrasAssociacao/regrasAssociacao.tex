\begin{Slide}{Mineração de itens frequentes}
\begin{itemize}
\item Dado:
\begin{itemize}
\item um conjunto $A = \{a_{1}, \cdots, a_{m}\}$ de itens,
\item uma tabela $T = (t_{1}, \cdots, t_{n})$ de transações sobre $A$,
\item um número $\beta_{min}$ que $0 < \beta_{min} \leq 1$, o
  \textbf{suporte mínimo}.
\end{itemize}

\item Objetivo 1:
\begin{itemize}
\item encontrar o conjunto de \textbf{itens frequentes}, tais que o
  \textbf{suporte} de cada conjunto de itens é maior ou igual ao $\beta_{min}$
  definido pelo usuário.
\end{itemize}
\end{itemize}
\end{Slide}

\begin{Slide}{Suporte e Confiança}

O suporte de um conjunto de itens $Z$, $suporte(Z)$, representa a
porcentagem de transações na base de dados que contêm os itens de
$Z$.

O suporte de uma regra de associação $A \rightarrow B$, $suporte(A
\rightarrow B)$, é dado por $suporte(A \cup B)$. 

\begin{equation}
confianca(A \rightarrow B) = \frac{P(A \cup B)}{P(A)} = \frac{suporte(A
\cup B)}{suporte(A)}
\end{equation}
\end{Slide}

\begin{Slide}{Heurísticas para Seleção de Regras de Associação}
\begin{itemize}
\item \textit{Lift}: um valor de \textit{lift} para uma regra ($A
  \rightarrow B$) superior a 1 indica que $A$ e $B$ aparecem mais
  frequentemente juntos do que o esperado, isso significa que a
  ocorrência de $A$ tem um efeito positivo sobre a ocorrência de $B$.
\item \textit{Convicção}: este valor indica que a probabilidade do
  item $A$ ocorrer sem o item $B$ é $X$ vezes menor.
\end{itemize}
\end{Slide}
